1)SELECT AVG(TIMESTAMPDIFF(YEAR, created_at, NOW())) AS avg_year FROM users;

2)SELECT DAYNAME(CONCAT(DATE_FORMAT(NOW(), '%Y'), DATE_FORMAT(created_at, '-%m-%d'))) `day of week`, COUNT(DAYNAME(CONCAT(DATE_FORMAT(NOW(), '%Y'), DATE_FORMAT(created_at, '-%m-%d')))) `count days` FROM users GROUP BY `day of week`;
