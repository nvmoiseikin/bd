1)SELECT gender 
	FORM profiles 
	INNER JOIN likes
	ON profiles.user_id = likes.user_id
	GROUP BY profiles.gender
	ORDER BY count(*)
    	LIMIT 1;

2)SELECT COUNT(*) 
	FROM likes 
	INNER JOIN profiles
	ON likes.target_id = profiles.user_id
  	AND likes.target_type_id = 2 
	ORDER BY profiles.birthday DESC 
	LIMIT 10;

3)SELECT CONCAT(first_name, ' ', last_name) AS user, COUNT(*) AS overall_activity
	FROM users
	INNER JOIN likes
	INNER JOIN media
	INNER JOIN messages
	ON likes.target_id = users.id
  	AND likes.target_type_id = 2
	AND media.user_id = users.id 
	AND messages.from_user_id = users.id
	GROUP BY users.id
	ORDER BY overall_activity
	  LIMIT 10;
